194// runtime profiler
195constexpr u16 PROFILER_FRAMES_RELEVANT_AVERAGE = 300;
196struct RuntimeProfilerData
197{
198 const char* name;
199 std::chrono::steady_clock::time_point last;
200 f64 measurements[PROFILER_FRAMES_RELEVANT_AVERAGE] = { 0 };
201 u16 head = 0;
202};
203static inline void profiler_tick(RuntimeProfilerData* data)
204{
205 data->measurements[data->head] = (std::chrono::steady_clock::now()-data->last).count()*MATH_CONVERSION_MS;
206 data->head = (data->head+1)%PROFILER_FRAMES_RELEVANT_AVERAGE;
207 data->last = std::chrono::steady_clock::now();
208}
209static inline f64 profiler_average(RuntimeProfilerData* data)
210{
211 f64 sum = .0;
212 for (u16 i=0;i<PROFILER_FRAMES_RELEVANT_AVERAGE;i++) sum += data->measurements[i];
213 return sum/PROFILER_FRAMES_RELEVANT_AVERAGE;
214}
